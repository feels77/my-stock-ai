## 🔑 핵심 데이터 구조

### Asset 모델
```js
{
  id: 'asset_1',          // 고유 식별자 (불변)
  group: '성장형',         // 자산군 (5종)
  account: 'ISA',         // 계좌명
  name: 'TIGER 미국나스닥', // 종목명
  ticker: '133690',       // 종목코드 (선택)
  quantity: 10,           // 보유수량
  avgPrice: 15000,        // 평균단가 (원)
  currentPrice: 17000,    // 현재가 (원)
}
```

### Transaction 모델
```js
{
  id: 'tx_1234567890',    // 고유 식별자
  date: '2026-06-12',     // 거래일자
  type: '매수',            // '매수' | '매도'
  assetId: 'asset_1',     // Asset.id 참조 (연결키)
  quantity: 5,             // 거래수량
  price: 16000,            // 거래단가
  memo: '월급날 적립',     // 메모 (선택)
}
```

### useAssets() 공개 인터페이스
```js
const {
  assets, transactions,          // 상태
  addAsset, removeAsset,         // 자산 CRUD
  updateAsset,                   // 자산 수정 (Phase 1.5B 추가)
  buyAsset, sellAsset,           // 매수/매도
  removeTrade, clearTransactions,// 거래 관리
  updateAllPrices,               // 현재가 업데이트 (Mock)
  isUpdatingPrices, lastUpdated, // 업데이트 상태
  totalAssetValue, totalCost,    // 포트폴리오 합계
  totalPnL, totalPnLPct,         // 손익 합계
  getGroupSummary,               // 자산군별 평가금액/비중 (Phase 2 추가)
} = useAssets();
```
